CREATE TABLE Volunteers
( 	
	PRIMARY KEY(user_id),
    	FOREIGN KEY (user_id) REFERENCES Users(user_id),
	affiliation CHAR(20),
	canPickUp BOOLEAN,
	canHomeShare BOOLEAN, 
	passengers SMALLINT UNSIGNED,
	suitcases SMALLINT UNSIGNED,
	beginHomeShare DATETIME,
	endHomeShare DATETIME
);

CREATE TABLE Contacts
(
	Contact_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	isVolunteer BOOLEAN,
	FOREIGN KEY (user_id) REFERENCES Users(user_id),
	contactName VARCHAR(50),
	contactPhone VARCHAR(15),
	contactRelation VARCHAR(30)
);

CREATE TABLE VolAvailables
( 	
	Avail_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	Volunteer_id INT NOT NULL,
	beginTime DATETIME,
	endTime DATETIME,
	filled BOOLEAN
);

CREATE TABLE Students
(
	PRIMARY KEY(user_id),
    	FOREIGN KEY (user_id) REFERENCES Users(user_id),
	affiliation VARCHAR(20),
	needPickUp BOOLEAN,
	needHomeShare BOOLEAN,
	beginHomeShare DATETIME,
	endHomeShare DATETIME,
	airline VARCHAR(25),
	flightNumber VARCHAR(25),
	arrivalTime DATETIME
);

CREATE TABLE PickUps
(
	PickUp_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	Student_id INT NOT NULL,
	Volunteer_id INT NOT NULL,
	date DATETIME
);

CREATE TABLE HomeShares
(
	HomeShare_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	FOREIGN KEY (vol_id) REFERENCES Volunteers(user_id),
	FOREIGN KEY (stu_id) REFERENCES Students(user_id)
);

CREATE TABLE Users
(
	Users_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	email VARCHAR(35),
	password VARCHAR(15),
	user_type VARCHAR(15),
	fName VARCHAR(20),
	lName VARCHAR(20),
	address VARCHAR(100),
	zip SMALLINT,
	phone VARCHAR(15),
	isMale BOOLEAN
);
